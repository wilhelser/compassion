def contractor_review_notification(contractor, user, project, review)
  @user = user
  @review = review
  @contractor = contractor
  @project = project
  mail(to: @contractor.email, bcc: "wil@wilhelser.com", subject: "#{@contractor.owner_first_name}, you've received a review!")
end
